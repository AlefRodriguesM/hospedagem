### DB ###
CREATE DATABASE db_hospedagem
;
CREATE TABLE `TB_USERS` (
  `PK_ID` INT(11) NOT NULL AUTO_INCREMENT,
  `DS_LOGIN` VARCHAR(60) DEFAULT NULL,
  `DS_PASSWORD` VARCHAR(256) NOT NULL,
  `TG_ADMIN` INT,
  `TG_INACTIVE` INT,
  `DT_REGISTER` TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP,
  PRIMARY KEY (`PK_ID`)
) ENGINE=INNODB AUTO_INCREMENT=4 DEFAULT CHARSET=LATIN1;

### SO ###
Va até "C:\Windows\System32\drivers\etc\hosts"
Edite e adicione "127.0.0.1		www.hospedagem.com.br"

Va até "C:\xampp\apache\conf\extra\httpd-vhosts"
Edite e adicione
<VirtualHost *:80>
    ServerAdmin webmaster@hospedagem.com.br
    DocumentRoot "C:/xampp/htdocs/hospedagem"
    ServerName www.hospedagem.com.br
    ErrorLog "logs/dummy-host2.example.com-error.log"
    CustomLog "logs/dummy-host2.example.com-access.log" common
	<Directory "C:/xampp/htdocs/hospedagem">
        Require all granted

        RewriteEngine On

        RewriteCond %{REQUEST_FILENAME} !-d
        RewriteCond %{REQUEST_FILENAME} !-f
        RewriteRule ^ index.php [QSA,L]
	</Directory>
</VirtualHost>
